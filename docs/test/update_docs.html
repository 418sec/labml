file_name = '../lab_getting_started.py'
content = open(file_name, 'r').read()
lines = content.split('\n')


MULTI_LINE_COMMENT = '"""'
AUTO_GENERATED = '+++++'

header = []
doc = []

if lines[0].startswith("#!"):
    header.append(lines.pop(0))

is_doc = False
while len(lines) > 0:
    l = lines.pop(0)
    l_s = l.strip()

    if not is_doc:
        if l_s == "":
            header.append(l)
        elif l_s.startswith(MULTI_LINE_COMMENT):
            if len(l_s) > 3 and l_s.endswith(MULTI_LINE_COMMENT):
#

single line comment

                doc.append(MULTI_LINE_COMMENT)
                doc.append(l_s[3:-3])
                doc.append(MULTI_LINE_COMMENT)
                break
            doc.append(l)
            is_doc = True
        else:
#

Code was found before documentation

            break
    else:
        doc.append(l)

        if l.startswith(MULTI_LINE_COMMENT):
            break

if len(doc) == 0:
    doc.append(MULTI_LINE_COMMENT)
    doc.append(MULTI_LINE_COMMENT)

assert(len(doc) >= 2)

doc_top = []
is_generated = False

while len(doc) > 0:
    l = doc.pop(0)
    l_s = l.strip()

    if not is_generated:
        if l_s == AUTO_GENERATED:
            is_generated = True
        else:
            doc_top.append(l)
    else:
        if l_s == AUTO_GENERATED:
            break

if len(doc) == 0:
    doc.append(doc_top.pop())

generated = []
if not is_generated:
    generated.append("")
generated.append(AUTO_GENERATED)
generated.append("Varuna created this")
generated.append(AUTO_GENERATED)

res = header + doc_top + generated + doc + lines

with open(file_name, 'w') as file:
    file.write("\n".join(res))
