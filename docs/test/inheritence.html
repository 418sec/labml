from lab import colors
from lab.logger import Logger

logger = Logger()
# 
class Ppo:
# 
    def __init__(self):
        logger.log("PPO: Init")
        self.model = 'MODEL'
        self.play_model: str = ""
        self.train_model: str = ""
# 
    def init_play(self):
        logger.log("PPO: Init play")
        self.play_model = "PLAY"
# 
    def init_train(self):
        logger.log("PPO: Init Train")
        self.train_model = "TRAIN"
# 
    def play(self):
        self.init_play()
        logger.log(f"PLAY: {self.model} {self.play_model}")
# 
    def train(self):
        self.init_train()
        logger.log(f"TRAIN: {self.model} {self.train_model}")
# 
class PpoRnd(Ppo):
# 
    def __init__(self):
        super().__init__()
        self.model += "_RND"
# 
    def init_play(self):
        super().init_play()
        self.play_model += "_RND"
# 
    def init_train(self):
        super().init_train()
        self.train_model += "_RND"
# 
    def run(self):
        super().init_play()
        super().init_train()
        logger.log(f"RUN: {self.model} {self.play_model}")
        logger.log(f"RUN: {self.model} {self.train_model}")


logger.log("RND PLAY", color=colors.BrightColor.cyan)
PpoRnd().play()

logger.log("RND TRAIN", color=colors.BrightColor.cyan)
PpoRnd().train()

logger.log("RND RUN", color=colors.BrightColor.cyan)
PpoRnd().run()

logger.log("PLAY", color=colors.BrightColor.cyan)
Ppo().play()

logger.log("TRAIN", color=colors.BrightColor.cyan)
Ppo().train()
